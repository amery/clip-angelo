Example:

 oDep := coDepository():new("MY00101")
 oDict := oDep:dictionary()
 currency := oDict:classBodyByName("currency")
 
 * select all objects of currency class whose attribute <CODE> begin with "U"
 idList:= codb_idList(currency:id,,,'code="U"')
 while !idList:eof()
 obj := idList:getValue()
 ? obj:code, obj:name
 idList:skip()
 end
